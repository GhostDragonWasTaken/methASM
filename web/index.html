import "std/math";

function factorial(n: int32) -> int32 {
  if (n <= 1) {
    return 1;
  }
  return n * factorial(n - 1);
}

function main() -> int32 {
  var result: int32 = factorial(10);
  println_int(result);
  return 0;
}
struct Point {
  x: float64;
  y: float64;
  
  function distance(other: Point*) -> float64 {
    var dx: float64 = this.x - other->x;
    var dy: float64 = this.y - other->y;
    return sqrt(dx*dx + dy*dy);
  }
}

function main() -> int32 {
  var p1: Point = {0.0, 0.0};
  var p2: Point = {3.0, 4.0};
  var dist: float64 = p1.distance(&p2);
  return 0;
}
import "std/net";
import "std/io";

function main() -> int32 {
  if (net_init() != 0) {
    puts("Failed to initialize network");
    return 1;
  }
  
  var sock: int64 = socket_tcp();
  var addr: cstring = sockaddr_in_any(8080);
  
  if (bind(sock, addr, 16) != 0) {
    puts("Bind failed");
    return 1;
  }
  
  listen(sock, 5);
  puts("Server listening on port 8080");
  
  // Accept loop...
  return 0;
}
